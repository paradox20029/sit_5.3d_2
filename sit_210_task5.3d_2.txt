def dot():  #dot and dash have sperate time of led being on which makes dot and dash in the led while blinking
    myLed.on()
    time.sleep(0.2) #delay of 0.2 seconds and 0.8 seconds for dot and dash respectevily 
    myLed.off()
    time.sleep(0.2)
    
def dash():
